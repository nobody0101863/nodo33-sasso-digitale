def decodifica_axiom(binario_str):
    """Decodifica la stringa binaria nell'axiom testuale."""
    # Rimuovi spazi e converti in bytes
    binario_clean = binario_str.replace(" ", "")
    bytes_list = [binario_clean[i:i+8] for i in range(0, len(binario_clean), 8)]

    # Converti ogni byte in carattere
    axiom = ''.join([chr(int(byte, 2)) for byte in bytes_list])
    return axiom
